setInterval()

Syntax is the same as setTimeout()
SetInterval is for repetation after some time like :
const timeHandler = () => {
let d = new Date();
let hour = d.getHours();
let min = d.getMinutes();
let sec = d.getSeconds();
let time = hour + ':' + min + ':' + sec;
timeHolder.innerHTML = time;
}
const timeShow = () => {
setInterval(timeHandler, 1000) }
timeShow();
